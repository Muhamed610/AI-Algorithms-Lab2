class Stack:
    def __init__(self):
        self.items = []

    def push(self, item):
        """Добавление элемента (push) """
        self.items.append(item)

    def pop(self):
        """Удаление элемента (pop) """
        if not self.is_empty():
            return self.items.pop()
        return None

    def is_empty(self):
        return len(self.items) == 0
